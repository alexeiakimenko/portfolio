SELECT FIO,Kafedra,Decanat
FROM Kafedra,Student
WHERE Kafedra IN
    (SELECT Kafedra
    FROM G
    WHERE [Group] IN
        (SELECT [Group]
        FROM Student
        WHERE ID_St IN
            (SELECT ID_st
            FROM Exzamen
            WHERE Predmet LIKE "Социология"
            )
        )
    )

3.	Выведите кафедры, список студентов, список студентов в алфавитном порядке.
SELECT s.FIO,
    (SELECT g.Kafedra
    FROM G g
    WHERE s."Group"=g."Group") AS "Кафедра"
FROM Student s
ORDER BY s.FIO
